createOptionsString(){
  var buffer = new StringBuffer();
  if (caseInsensitive === true){
    buffer.add("i");
  }
  if (multiLine === true){
    buffer.add("m");
  }    
  if (dotAll === true){
    buffer.add("s");
  }    
  if (extended === true){
    buffer.add("x");
  }    
  options = buffer.toString();
}
