 SelectorBuilder inRange(String fieldName, min, max, {bool minInclude: true, bool maxInclude: false}) {
   Map rangeMap = {};
   if (minInclude){
     rangeMap["\$gte"] = min;
   }
   else{
     rangeMap["\$gt"] = min;
   }
   if (maxInclude){
     rangeMap["\$lte"] = max;
   }
   else{
     rangeMap["\$gt"] = max;
   }
   map[fieldName] = rangeMap;
   return this;
 }
