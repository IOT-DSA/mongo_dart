Code Binary serialize() #

Binary serialize(){
  Binary buffer = new Binary(messageLength);
  writeMessageHeaderTo(buffer);
  buffer.writeInt(0);
  _collectionFullName.packValue(buffer);
  buffer.writeInt(flags);
  _selector.packValue(buffer);
  _document.packValue(buffer);
  buffer.offset = 0;
  return buffer;
}
